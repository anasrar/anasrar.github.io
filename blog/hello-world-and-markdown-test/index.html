class FizzBuzz {
  public static void main(String[] args) {
    for (int i = 1; i <= 100; i++) {
      String r = "";
      Boolean fizz = i % 3 == 0;
      Boolean buzz = i % 5 == 0;
      if(fizz){
        r += "fizz";
      }
      if(buzz){
        r += "buzz";
      }
      System.out.println((r == "") ? i : r);
    }
  }
}
